def calculator():
    current_value = 0
    print("Simple Calculator")
    print("Operations:")
    print("+ : Addition")
    print("- : Subtraction")
    print("* : Multiplication")
    print("/ : Division")
    print("v : Reset calculator")
    print("u : Quit")

    while True:
        operation = input("\nEnter an operation: ").strip()
        
        if operation == 'u':
            print("Exiting the calculator. Goodbye!")
            break
        elif operation == 'v':
            current_value = 0
            print(f"Calculator reset. Current value: {current_value}")
        elif operation in ['+', '-', '*', '/']:
            try:
                number = float(input("Enter a number: ").strip())
                if operation == '+':
                    current_value += number
                elif operation == '-':
                    current_value -= number
                elif operation == '*':
                    current_value *= number
                elif operation == '/':
                    if number != 0:
                        current_value /= number
                    else:
                        print("Error: Division by zero is not allowed.")
                        continue
                print(f"Current value: {current_value}")
            except ValueError:
                print("Invalid input. Please enter a valid number.")
        else:
            print("Invalid operation. Please try again.")

if __name__ == "__main__":
    calculator()
