# 1. define a function "sum_main_diagonal" that takes square matrix as input

def sum_main_diagonal(matrix):

    # we need to store the sum of main diagonal of matrix

    diagonal_sum = 0

    # we use cycle to go through the elements along the main diagonal of the matrix.
    # the index "i" is used to access the diagonal element at position "i,i"

    for i in range(len(matrix)):

        # we add the value of each diagonal element to the "diagonal sum"

        diagonal_sum += matrix[i][i]

    return diagonal_sum

    # after calculating the sum, we return it.

# example:

matrix = [
    [1, 2, 3],
    [4, 5, 6],
    [7, 8, 9]
]
diagonal_sum = sum_main_diagonal(matrix)
print("the sum of numbers along the main diagonal is: ", diagonal_sum)

# we go through the elements along the main diagonal of the matrix
# for each element, we add its value to the sum.
# after going through all diagonal elemnts, we return the sum, which represents the sum of numbers along the main diagonal of the matrix.
