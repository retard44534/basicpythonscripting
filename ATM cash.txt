# define a function " dispense_cash" that takes the amount to be dispensed as input.

ddef atm():
    denominations = [100, 50, 20, 10]
    
    print("ATM Machine")
    print("Available denominations: 10€, 20€, 50€, 100€")
    
    while True:
        try:
            amount = int(input("Enter the amount you want to withdraw (or 0 to quit): ").strip())
            if amount == 0:
                print("Exiting the ATM. Goodbye!")
                break
            if amount % 10 != 0:
                print("The amount must be a multiple of 10€. Please try again.")
                continue

            banknotes = {}
            for denomination in denominations:
                count = amount // denomination
                if count > 0:
                    banknotes[denomination] = count
                amount = amount % denomination
            
            if amount == 0:
                print("The ATM will dispense the following banknotes:")
                for denomination, count in banknotes.items():
                    print(f"{count} x {denomination}€")
            else:
                print("The amount cannot be dispensed with available denominations.")
        except ValueError:
            print("Invalid input. Please enter a valid amount.")

if __name__ == "__main__":
    atm()
