# define a function "count_numbers" that takes a list of 10 numbers as input

def count_numbers(numbers):

    # initialize counters to keep a track of the positive, negative and zero values.

    positive_count = 0
    negative_count = 0
    zero_count = 0

    # go through the list# we go through each number in the list and cheeck if it is positive, negative or zero 

    for number in numbers:
        if number > 0:
            positive_count += 1
    elif number < 0:
        negative_count += 1
    else:
        zero_count += 1

    # after going through the list we return the counts of positive, negative and zero values.

    return positive_count, negative_count, zero_count


# Input 10 numbers from the user and store them in a list

numbers = []
for i in range(10):
    num = float(input(f"Enter number {i + 1}: "))
    numbers.append(num)

# Count the numbers
positive_count, negative_count, zero_count = count_numbers(numbers)
# we call the "count_numbers" function to get the counts of positive, negative, and zero values.

# Print the results
print("Number of positive values: ", positive_count)
print("Number of negative values: ", negative_count)
print("Number of zero values: ", zero_count)
